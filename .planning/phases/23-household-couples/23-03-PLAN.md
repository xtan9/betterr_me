**1. Create lib/hooks/use-household.ts:**

```typescript
"use client";
import useSWR from "swr";
import { fetcher } from "@/lib/fetcher";
import { useState } from "react";

interface HouseholdResponse {
  household_id: string;
  role: "owner" | "member";
  members: HouseholdMemberWithProfile[];
  invitations: HouseholdInvitation[];
}

export function useHousehold() {
  const { data, error, mutate } = useSWR<HouseholdResponse>(
    "/api/money/household",
    fetcher
  );
  const [viewMode, setViewMode] = useState<"mine" | "household">("mine");

  const isLoading = !data && !error;
  const isMultiMember = (data?.members?.length ?? 0) > 1;

  return {
    householdId: data?.household_id ?? null,
    members: data?.members ?? [],
    invitations: data?.invitations ?? [],
    isMultiMember,
    isOwner: data?.role === "owner",
    isLoading,
    error,
    mutate,
    viewMode,
    setViewMode,
  };
}
```

The view mode state lives in this hook. All money pages that need view switching will call `useHousehold()` and pass `viewMode` to their data hooks.

**2. Create components/money/household-view-tabs.tsx:**

"use client" component using shadcn Tabs:
```tsx
import { Tabs, TabsList, TabsTrigger } from "@/components/ui/tabs";
import { useTranslations } from "next-intl";
import type { ViewMode } from "@/lib/db/types";

interface Props {
  value: ViewMode;
  onValueChange: (v: ViewMode) => void;
  isMultiMember: boolean;
}

export function HouseholdViewTabs({ value, onValueChange, isMultiMember }: Props) {
  const t = useTranslations("money.household");
  if (!isMultiMember) return null;

  return (
    <Tabs value={value} onValueChange={(v) => onValueChange(v as ViewMode)}>
      <TabsList>
        <TabsTrigger value="mine">{t("tabMine")}</TabsTrigger>
        <TabsTrigger value="household">{t("tabHousehold")}</TabsTrigger>
      </TabsList>
    </Tabs>
  );
}
```
