    ```typescript
    async getWorkoutsWithSummary(userId: string, options?: { limit?: number; offset?: number }) {
      const limit = options?.limit ?? 20;
      const offset = options?.offset ?? 0;
      const { data, error } = await this.supabase
        .from("workouts")
        .select(`
          *,
          workout_exercises (
            id,
            exercise:exercises (name),
            sets:workout_sets (weight_kg, reps, is_completed, set_type)
          )
        `)
        .eq("user_id", userId)
        .eq("status", "completed")
        .order("started_at", { ascending: false })
        .range(offset, offset + limit - 1);
      if (error) throw error;
      return (data ?? []).map(w => ({
        id: w.id, title: w.title, notes: w.notes,
        started_at: w.started_at, completed_at: w.completed_at,
        duration_seconds: w.duration_seconds,
        exerciseCount: w.workout_exercises.length,
        exerciseNames: w.workout_exercises.map((we: any) => we.exercise?.name).filter(Boolean),
        totalVolume: w.workout_exercises.reduce((sum: number, we: any) =>
          sum + we.sets.filter((s: any) => s.is_completed)
            .reduce((v: number, s: any) => v + (s.weight_kg ?? 0) * (s.reps ?? 0), 0), 0),
        totalSets: w.workout_exercises.reduce((sum: number, we: any) =>
          sum + we.sets.filter((s: any) => s.is_completed).length, 0),
      }));
    }
    ```

    **Step 3: Add `getExerciseHistory()` to WorkoutsDB.**
    Fetches per-workout aggregated stats for a single exercise across all completed workouts. Accepts optional `since` ISO date for date range filtering:
    ```typescript
    async getExerciseHistory(exerciseId: string, userId: string, options?: { since?: string }): Promise<ExerciseHistoryEntry[]> {
      let query = this.supabase
        .from("workout_exercises")
        .select(`
          workout_id,
          sets:workout_sets (weight_kg, reps, duration_seconds, is_completed, set_type),
          workout:workouts!inner (id, started_at, status, user_id)
        `)
        .eq("exercise_id", exerciseId)
        .eq("workout.status", "completed")
        .eq("workout.user_id", userId);
      if (options?.since) {
        query = query.gte("workout.started_at", options.since);
      }
      const { data, error } = await query.order("workout.started_at", { referencedTable: "workouts", ascending: true });
      if (error) throw error;
      return (data ?? []).map((row: any) => {
        const completedNormal = row.sets.filter((s: any) => s.is_completed && s.set_type === "normal");
        return {
          date: row.workout.started_at.split("T")[0],
          workout_id: row.workout.id,
          best_set_weight_kg: completedNormal.length > 0 ? Math.max(...completedNormal.map((s: any) => s.weight_kg ?? 0)) || null : null,
          best_set_reps: completedNormal.length > 0 ? Math.max(...completedNormal.map((s: any) => s.reps ?? 0)) || null : null,
          total_volume: completedNormal.reduce((sum: number, s: any) => sum + (s.weight_kg ?? 0) * (s.reps ?? 0), 0) || null,
          total_sets: completedNormal.length,
        };
      });
    }
    ```
    Import `ExerciseHistoryEntry` type from `@/lib/db/types` at the top of the file.
