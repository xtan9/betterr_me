    ```typescript
    import useSWR from "swr";
    import { fetcher } from "@/lib/fetcher";

    export function useAccounts() {
      const { data, error, isLoading, mutate } = useSWR(
        "/api/money/accounts",
        fetcher,
        { keepPreviousData: true }
      );
      return { data, error, isLoading, mutate };
    }
    ```
