**3. Update `lib/plaid/sync.ts`** — Apply merchant rules during sync:

Add a new function `applyMerchantRules`:

```typescript
async function applyMerchantRules(
  transactions: Array<{ merchant_name: string | null; category: string | null; [key: string]: unknown }>,
  householdId: string,
  supabaseAdmin: SupabaseClient
): Promise<void> {
  const merchantNames = [...new Set(
    transactions.map(t => t.merchant_name).filter(Boolean)
  )] as string[];

  if (merchantNames.length === 0) return;

  // Import MerchantRulesDB or query directly with admin client
  const { data: rules } = await supabaseAdmin
    .from("merchant_category_rules")
    .select("merchant_name_lower, category_id, categories(name)")
    .eq("household_id", householdId)
    .in("merchant_name_lower", merchantNames.map(n => n.toLowerCase()));

  if (!rules?.length) return;

  const ruleMap = new Map(rules.map(r => [r.merchant_name_lower, r]));

  for (const txn of transactions) {
    if (txn.merchant_name) {
      const rule = ruleMap.get(txn.merchant_name.toLowerCase());
      if (rule) {
        txn.category = (rule.categories as { name: string }).name;
        (txn as Record<string, unknown>).category_id = rule.category_id;
      }
    }
  }
}
```

Call `applyMerchantRules` AFTER converting Plaid transactions but BEFORE upserting into the DB (both for `validAdded` and `validModified` arrays). This ensures merchant rules override Plaid's default categorization.

Also update the insert mapping in sync.ts to include `category_id` field from the transaction objects (currently only maps `category` TEXT).
